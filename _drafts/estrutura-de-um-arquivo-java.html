Abaixo podemos ver um exemplo de um arquivo Java válido.

package com.edgardleal.ocjp;

import java.util.List;

/**
 * Classe publica do arquivo. Pode ser acessada por outras classes.
 */
public class Main {
    private List instancia;
    public static List statico;

    public static void main(String[] args) {
        estatico = null; // Não preciso do nome da classe
        Main.estatico = null;  
        System.out.println("Hello World!"); // Display the string.
    }
}

// Classe interna do arquivo, e so pode ser visualizada dentro deste arquivo.
class Interna {
    String nome;
}
